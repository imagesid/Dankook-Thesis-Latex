\chapter{본\quad 론}

본론은 1\textasciitilde3개 내외의 장으로 구성하며, 연구내용, 연구방법 및 연구결과를 기술한다. 서론에서 정의한 문제(연구질문)에 대한 답을 찾아가는 과정을 논리적이고 일목요연하게 기술한다.

\section{연구방법}

\text{연구방법의 기능(function)}
\begin{itemize}
    \item 서론에 기술한 연구질문에 답을 하기 위해 연구자가 무엇을 했는지 자세하게 기술하여 독자에게 알려줌
\end{itemize}

\text{연구방법의 내용(content)}
\begin{enumerate}
    \item 재료 (연구대상)
    \item 방법: 해석적(analytic), 실험적(experimental), 수치해석적(numerical), 통계적(statistical) 등 연구에서 활용한 방법을 구체적으로 기술
\end{enumerate}

\text{연구방법의 시제(verb tense)}
\begin{enumerate}
    \item 시행했던 방법인 경우 과거형 (\ldots 를 OO방법으로 측정했다)
    \item 데이터 (자료)를 기술할 때는 현재형 (\ldots 는 평균과 표준편차로 요약된다)
\end{enumerate}

\section{연구결과}

\text{연구결과의 기능(function)}
\begin{enumerate}
    \item 연구방법에서 기술했던 방법에 의해 얻은 결과에 대해 자세하게 기술
    \item 연구결과를 뒷받침하는 근거 (데이터)를 제시하거나 독자에게 직접적으로 표(Tables)나 그림(Figures)으로 근거를 뒷받침
\end{enumerate}

\text{연구결과에 포함할 내용(content)}
\begin{enumerate}
    \item 서론에서 제기한 연구질문에 적절한 결과만 보고 (해석하지 않고 데이터만 제시 -- 결과 해석은 IV.\ 토론에서)
    \item 연구자의 연구가설을 지지 여부에 관계없이 모든 결과를 포함
\end{enumerate}

\text{연구결과의 시제(verb tense)}
\begin{itemize}
    \item 과거형으로 결과 보고
\end{itemize}
